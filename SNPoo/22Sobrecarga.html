Sobrecarga

Definición de sobrecarga. En Java es posible definir dos o más métodos dentro de la misma clase que tengan el mismo nombre, pero con sus listas de parámetros distintas. Cuando ocurre esto, se dice que los modos están sobrecargados y a este proceso se le denomina sobrecarga de modo.


Como hace la selección. Cuando se invoca un método sobrecargado, Java utiliza el tipo y/o el número de argumentos como guía para determinar la versión del método sobrecargado que realmente debe llamar.


Programa 1.


        class Sobrecarga {
          void prueba() {
            System.out.println("Sin parametros");
          }
          void prueba(int a) {
            System.out.println("a: " + a);
          }
          void prueba(int a, int b) {
            System.out.println("a y b: " + a + " " + b);
          }
          double prueba(double a) {
            System.out.println("a double: " + a);
            return a + a;
          }
         }

         public class SobrecargaDemo {
           public static void main(String args[]) {
             Sobrecarga ob = new Sobrecarga();
             double resultado;
    
             ob.prueba();
             ob.prueba(10);
             ob.prueba(10,20);
             resultado = ob.prueba(123.2);
             System.out.println("Resultado de ob.prueba(123.2): " + resultado);
            }
          }

Programa 2.


        class Sobrecarga1 {
          void prueba() {
            System.out.println("Sin parametros");
          }
          void prueba(int a, int b) {
            System.out.println("a y b: " + a + " " + b);
          }
          void prueba(double a) {
            System.out.println("Dentro de prueba(double) a: " + a);
          }

         public class SobrecargaDemo1 {
           public static void main(String args[]) {
             Sobrecarga1 ob = new Sobrecarga1();
             int i = 8;
    
             ob.prueba();
             ob.prueba(10, 20);
     
             ob.prueba(i);
             ob.prueba(123.2);
           }
         }

Programa 3.


        class Altura {
          double altura;
          String unidades;

          public void setAltura(double altura) {
            this.altura = altura;
            this.unidades = "cm";
          }

          public void setAltura(double altura, String unidades) {
            this.altura = altura;
            this.unidades = unidades;
          }

          public void imprimir() {
            System.out.println(this.altura + " " + this.unidades);
          }
        } 

        public class AlturaDriver {
          public static void main(String[] args) {
            Altura miAltura = new Altura();
            miAltura.setAltura(72.0, "in");
            miAltura.imprimir();
            miAltura.setAltura(180.0);
            miAltura.imprimir();
          } 
        }
