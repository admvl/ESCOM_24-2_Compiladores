                // mover al frente de la lista
                q := cabeza;
                cabeza := p↑.sig;
                p↑.sig := p↑.sig↑.sig;
                cabeza↑.sig := q;
                busqueda := cabeza;
                goto 999 // cortar
                end

      j = i;

procedimiento ordenamiento(var r: arreglo; l,u: entero);

var i,j: entero;
    tem: arreglo;
    
begin
while u > l do begin
    if r[i].k > r[i + 1].k then begin
        tem := r[i];
        r[i] := r[i + 1];
        r[i + 1] := tem;
        j := i
        end;
    u := j
    end
end;
