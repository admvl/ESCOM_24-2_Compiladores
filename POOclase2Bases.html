#include <iostream>

using namespace std;

class person
{
	char name[30];
	int age;
	
	public:
		void getdata(void);
		void display(void);
};

void person :: getdata(void)
{
	cout << "Agregar Nombre:" ;
	cin >> name;
	cout << "Agregar Edad:";
	cin >> age;
}

void person :: display(void)
{
	cout << "\nNombre:" << name;
	cout << "\nEdad:" << age;
}

int main()
{
	person p;

	p.getdata();
	p.display();

	return 0;
}

EJEMPLO DE PROGRAMA EN JAVA


public class Person {

	private String name;

	public void setName(String name) {

	this.name = name;

	}

	public String getName() {

	return this.name;

	}

	public void swapPerson(Person otherPerson) {

	String temp;
	temp = otherPerson.name;
	otherPerson.name = this.name;
	this.name = temp;

	} // Fin de swapPerson

	public static void main(String[] args) {

		Person person1 = new Person();
		Person person2 = new Person();
		person1.setName("Jonathan");
		person2.setName("Benji");
		System.out.println(person1.getName() + ", " +
			person2.getName());

		person1.swapPerson(person2);
		System.out.println(person1.getName() + ", " +
			person2.getName());

	} // Fin de main

} // Fin de clase Person

    Circle1 circle1 = new Circle1();
    System.out.println("El area del circulo de raio "
	+ circle1.radius + " es " + circle1.getArea() );
    // Crea un circulo con radio 25
    Circle1 circle2 = new Circle1(25);
    System.out.println("El area del circulo de radio "
	+ circle2.radius + " es " + circle2.getArea());
    // Crea un circulo de radio 125
    Circle1 circle3 = new Circle1(125);
    System.out.println("El area del circulo de raio "
	+ circle3.radius + " es " + circle3.getArea());
    // Modifica el radio del circulo
    circle2.radius = 100;
    System.out.println("El area del circulo de radio "
	+ circle2.radius + " es " + circle2.getArea() );
    }
  }
// Define la clase circulo con dos constructores
class Circle1 {
  double radius ;
  /** Construir del circulo con radio 1 */
  Circle1() {
    radius = 1.0;
  }
  /** Construir un circulo con radio especificado */
  Circle1(double newRadius) {
    radius = newRadius;
  }
  /** Regresa el area del circulo */
  double getArea() {
    return radius * radius * Math.PI;
  }
}
