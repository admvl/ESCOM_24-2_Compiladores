La clase Estudiante, con constructor y método String


public class Estudiante extends Persona {
	//Campos
	private int id;
	//Constructores
	public Estudiante() {
		super(); //Llama al padre del constructor
		id = -1;
	}
	public Estudiante(String nombre) {
		super(nombre);
		id = -1;
	}
	//Metodos
	public int getId() {
		return this.id;
	}
	public void setId(int elId) {
		this.id = elId;
	}
	public String toString() {
		return "Estudiante nombrado" +
			this.getNombre() +
			"Con id" + this.id;
	}
}

Estudiante b = new Estudiante("Beatriz");
System.out.println(b);
Estudiante nombrado Beatriz con id -1
