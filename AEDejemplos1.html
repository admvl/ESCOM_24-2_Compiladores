insertarOrdenar(int s[], int n) {
   int i, j;
   
   for(i = 1; i < n; i++) {
      j = i;
      
      while(j > 0 && s[j] < s[j - 1] ) {
         intercambiar(&s[j], &s[j - 1] )
         j = j - 1;
      }
   }
