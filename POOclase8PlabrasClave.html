public class fecha {
	public Fecha() {
		dia = 1;
		mes = 1;
		anio = 2018;
	}
	public fecha(int elDia, int elMes, int elAnio) {
		dia = elDia;
		mes = elMes;
		anio = elAnio;
	}
	public boolean equals (Object lder) {
		if(!(lder instanceof fecha))
			return false;
		fecha lderFecha = (fecha) lder;
		return lderFecha.dia == dia && lderFecha.mes == mes &&
			lderFecha.anio == anio;
	}
	public String toString() {
		return dia + "/" + mes + "/" + anio;
	}
	private int dia;
	private int mes;
	private int anio;
}

pubic fecha() {
	this(1,1,2018);
