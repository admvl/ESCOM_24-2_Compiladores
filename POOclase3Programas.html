#include <iostream>
#include <string>

using namespace std;

class Point {
  int i, j, k;
public:
  Point(): i(0), j(0), k(0) {}
  Point(int ii, int jj, int kk)
    : i(ii), j(jj), k(kk) {}
  void print(const string& msg = "") const {
    if(msg.size() != 0) cout << msg << endl;
    cout << "i = " << i << ", "
      << "j = " << j << ", "
      << "k = " << k << endl;
  }
};
int main() {
  Point p, q(1,2,3);
  p.print("value of p");
  q.print("value of q");
} ///:~

OTRO EJEMPLO DE PROGRAMA EN JAVA


public class Person {

	private String name;

	public void setName(String name) {

	this.name = name;

	}

	public String getName() {

	return this.name;

	}

	public void swapPerson(Person otherPerson) {

	String temp;
	temp = otherPerson.name;
	otherPerson.name = this.name;
	this.name = temp;

	} // Fin de swapPerson

	public static void main(String[] args) {

		Person person1 = new Person();
		Person person2 = new Person();
		person1.setName("Jonathan");
		person2.setName("Benji");
		System.out.println(person1.getName() + ", " +
			person2.getName());

		person1.swapPerson(person2);
		System.out.println(person1.getName() + ", " +
			person2.getName());

	} // Fin de main

} // Fin de clase Person

// Define la clase circulo con dos constructores
class Circle1 {
  double radius ;
  /** Construir del circulo con radio 1 */
  Circle1() {
    radius = 1.0;
  }
  /** Construir un circulo con radio especificado */
  Circle1(double newRadius) {
    radius = newRadius;
  }
  /** Regresa el area del circulo */
  double getArea() {
    return radius * radius * Math.PI;
  }
}
